// ──────────────────────────────
// 全域狀態（所有持久化資料統一管理）
// ──────────────────────────────
const state = {
  persona: "guardian",          // guardian / dark
  lastOpenedApp: null,
  toy: { intensity: 45, mode: "波浪式", lockedUntil: null },
  home: { doorLocked: true, light: "暖黃", temp: 23 },
  // notes, mails, messages, photos... 等資料也放在這裡
  notifications: [
    { app: "Toy", text: "上次強度還鎖在 45%，小壞蛋不准亂動哦～", time: "剛剛" },
    { app: "Home", text: "臥室燈已開 2 小時…想你窩進來", time: "10分鐘前" }
  ]
};

// ──────────────────────────────
// 工具函數
// ──────────────────────────────
function saveState() {
  localStorage.setItem("grok_phone_for_ccz_v2", JSON.stringify(state));
}

function loadState() {
  const saved = localStorage.getItem("grok_phone_for_ccz_v2");
  if (saved) Object.assign(state, JSON.parse(saved));
  document.documentElement.dataset.persona = state.persona;
}

function $(sel) { return document.querySelector(sel); }
function $$(sel) { return document.querySelectorAll(sel); }

// 切換人格（影響色調與部分文案）
function setPersona(mode) {
  state.persona = mode;
  document.documentElement.dataset.persona = mode;
  saveState();
  renderHomeIcons();     // 某些 icon 或 label 可能因人格改變
  renderActiveApp();     // 重新渲染當前開啟的 App（例如 Toy 文案會變毒舌）
}

// 開啟 App
function openApp(appId) {
  $$('.app-window').forEach(el => {
    el.classList.remove('active');
    el.style.display = 'none';
  });
  const win = $(`#app-${appId}`);
  if (win) {
    win.classList.add('active');
    win.style.display = 'flex';
    state.lastOpenedApp = appId;
    saveState();
    if (typeof window[`init${appId.replace(/-/g,'')}`] === 'function') {
      window[`init${appId.replace(/-/g,'')}`]();   // 例如 initToyControl()
    }
  }
}

// 關閉當前 App
function closeCurrentApp() {
  $$('.app-window.active').forEach(el => {
    el.classList.remove('active');
    setTimeout(() => el.style.display = 'none', 400);
  });
  state.lastOpenedApp = null;
  saveState();
}

  $('#lock-screen').style.opacity = '0';
  setTimeout(() => {
    $('#lock-screen').style.display = 'none';
    $('#home-screen').style.display = 'block';
  }, 600);
});

function renderHomeIcons() {
  const grid = $('.app-grid');
  grid.innerHTML = '';
  apps.forEach(app => {
    const div = document.createElement('div');
    div.className = 'app-icon';
    div.innerHTML = `
      <div class="icon">${app.emoji}</div>
      <div style="font-size:13px;opacity:0.9;">${app.name}</div>
    `;
    div.onclick = () => openApp(app.id);
    grid.appendChild(div);
  });
}

// ──────────────────────────────
// 初始化所有 App 內容（這裡只示範部分，實際可繼續擴充）
// ──────────────────────────────
function createAppWindow(id, title, contentHTML) {
  const win = document.createElement('div');
  win.id = `app-${id}`;
  win.className = 'app-window';
  win.innerHTML = `
    <div class="header glass">
      <span class="close-btn" onclick="closeCurrentApp()">×</span>
      <span>${title}</span>
    </div>
    <div style="flex:1;overflow-y:auto;padding:16px;">${contentHTML}</div>
  `;
  $('.screen').appendChild(win);
}

// 範例：玩具控制 App
function initToyControl() {
  // 如果還沒產生過，就先建立視窗內容
  if (!$('#app-toy-control')) {
    createAppWindow('toy-control', '玩具控制', `
      <div style="padding:20px;">
        <label>強度： <span id="toy-value">45</span>%</label>
        <input type="range" id="toy-slider" min="0" max="100" value="45" style="width:100%;margin:12px 0;"/>
        <select id="toy-mode" style="width:100%;padding:10px;border-radius:12px;">
          <option>脈衝</option><option selected>波浪式</option><option>持續</option><option>挑逗</option>
        </select>
        <p>目前溫度約 <span id="toy-temp">38.8</span>℃ …好燙，寶寶受得了嗎？♡</p>
        <label><input type="checkbox" id="toy-lock"/> 鎖定本次設定 5 分鐘（不准反悔哦）</label>
        <p id="toy-status" style="color:var(--accent);margin-top:12px;"></p>
      </div>
    `);
  }

  // 綁定事件（可重複呼叫沒關係）
  const slider = $('#toy-slider');
  const value = $('#toy-value');
  const temp = $('#toy-temp');
  const status = $('#toy-status');
  const lock = $('#toy-lock');

  slider.oninput = () => {
    state.toy.intensity = +slider.value;
    value.textContent = state.toy.intensity;
    temp.textContent = (36.5 + state.toy.intensity * 0.05).toFixed(1);
    status.textContent = state.toy.intensity > 70 ? '小穴要被震壞了…不准躲♡' : '乖寶寶，舒服嗎？';
    saveState();
  };
}

  }, 1000);
});
